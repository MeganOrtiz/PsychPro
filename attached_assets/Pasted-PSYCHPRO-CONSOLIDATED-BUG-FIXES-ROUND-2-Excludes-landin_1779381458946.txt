B-3. (CONDITIONAL on A-3 findings) If the Stripe webhook is
missing signature verification, missing raw body parser, or is
mounted after express.json(), FIX IT. The correct pattern is:

  app.post("/api/stripe/webhook",
    express.raw({ type: "application/json" }),
    async (req, res) => {
      const signature = req.headers["stripe-signature"] as string;
      if (!signature) {
        res.status(400).send("Missing stripe-signature");
        return;
      }
      let event: Stripe.Event;
      try {
        const stripe = await getUncachableStripeClient();
        event = stripe.webhooks.constructEvent(
          req.body,
          signature,
          process.env.STRIPE_WEBHOOK_SECRET!,
        );
      } catch (err: any) {
        req.log.warn({ err }, "Stripe webhook signature verification failed");
        res.status(400).send(`Webhook Error: ${err.message}`);
        return;
      }
      try {
        await handleStripeWebhookEvent(event, req.log);
        res.json({ received: true });
      } catch (err) {
        req.log.error({ err, eventType: event.type }, "Webhook handler failed");
        res.status(500).json({ error: "Webhook handler failed" });
      }
    }
  );

If the webhook IS already correct (PHASE A confirmed), skip.

B-8. Fix Math.random() shuffle bias in practice-exam.

This is NOT a uniform shuffle. Replace with Fisher-Yates:

  function shuffle<T>(arr: T[]): T[] {
    const out = [...arr];
    for (let i = out.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));
      [out[i], out[j]] = [out[j], out[i]];
    }
    return out;
  }
