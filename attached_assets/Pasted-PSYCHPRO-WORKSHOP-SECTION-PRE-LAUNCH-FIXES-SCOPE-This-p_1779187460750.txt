X-3.  ADD BREADCRUMBS TO DEEPER WORKSHOP PAGES
File: artifacts/neuronotes/src/components/breadcrumbs.tsx (NEW FILE)

Action: Create a reusable Breadcrumbs component.

    import { Link } from "wouter";
    import { ChevronRight } from "lucide-react";
    import { STUDY_PALETTE as P } from "@/lib/study-theme";

    export type BreadcrumbItem = {
      label: string;
      href?: string; // omit for current page
    };

    export function Breadcrumbs({ items }: { items: BreadcrumbItem[] }) {
      return (
        <nav
          aria-label="Breadcrumb"
          className="flex items-center gap-1.5 text-xs mb-4"
          data-testid="breadcrumbs"
        >
          {items.map((item, idx) => {
            const isLast = idx === items.length - 1;
            return (
              <span key={idx} className="flex items-center gap-1.5">
                {idx > 0 && (
                  <ChevronRight className="w-3 h-3"
                                style={{ color: P.mistSoft }} />
                )}
                {isLast || !item.href ? (
                  <span style={{ color: P.mist }} data-testid={`crumb-current`}>
                    {item.label}
                  </span>
                ) : (
                  <Link href={item.href}>
                    <span className="hover:underline cursor-pointer"
                          style={{ color: P.tealDeep }}
                          data-testid={`crumb-${idx}`}>
                      {item.label}
                    </span>
                  </Link>
                )}
              </span>
            );
          })}
        </nav>
      );
    }
